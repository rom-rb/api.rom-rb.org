def uri
  uri = config.fetch(:uri) { fail Error, '+uri+ configuration missing' }
  uri = URI(join_path(uri, path))
  if request_method == :get && params.any?
    uri.query = self.class.config.param_encoder.call(params)
  end

  uri
end
