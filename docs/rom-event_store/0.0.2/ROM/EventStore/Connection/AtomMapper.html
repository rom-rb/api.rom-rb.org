ALIASES = 
{
  eventId: :id,
  eventType: :type,
  eventNumber: :number,
  updated: :created_at
}.freeze
ATTRS = 
(ALIASES.values + [:data]).freeze
MAPPER = 
compose do |ops|
  ops << t(:symbolize_keys!)
  ops << t(:map_hash!, ALIASES)
  ops << t(:map_key!, :created_at, t(:to_time))
  ops << t(-> event { event.keep_if { |key| ATTRS.include?(key) } })
end
