def reset_data
  repo   = Rugged::Repository.new(@path)
  walker = Rugged::Walker.new(repo)
  branch = (@options || {}).fetch(:branch, 'refs/head/master')
  ref    = repo.references[branch]

  @connection = ref.log rescue []
end
