Overview

Implements the cassandra-specific Batch command

Examples:

class Batch < ROM::Cassandra::Batch
  relation :items
  register_as :batch

  def execute
    super {
      self
        .add(keyspace(:domain).table(:items).delete.where(id: 1))
        .add("INSERT INTO logs.items (id, text) VALUES (1, 'deleted');")
    }
  end
end
