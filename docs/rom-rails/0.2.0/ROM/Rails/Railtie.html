def self.disconnect
  return unless ROM.env

  ROM.env.repositories.each_value do |repository|
    repository.adapter.disconnect
  end
end

def self.load_files(type, root)
  Dir[root.join("app/#{type}/**/*.rb").to_s].each do |path|
    load(path)
  end
end
