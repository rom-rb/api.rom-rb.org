def set(error)
  case error
  when ActiveModel::Errors
    __setobj__ error
  when ROM::Model::ValidationError
    __setobj__ error.errors
  when nil
    # do nothing
  else
    add(:base, "a database error prevented saving this form")
  end

  self
end
