def configure(&block)
  config.rom = Configuration.new unless config.respond_to?(:rom)

  if block.arity == 1
    block.call(config.rom)
  else
    super
  end
end

def setup(gateways)
  raise(
    MissingGatewayConfigError,
    "seems like you didn't configure any gateways"
  ) unless gateways.any?

  ROM.setup(gateways)
end
