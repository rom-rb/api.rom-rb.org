def call(tuple, mapping = self.mapping)
  tuple.map do |key, value|
    case value
    when Hash  then [key, Hash[call(value, mapping[key])]]
    when Array then [key, value.map { |v| Hash[call(v, mapping[key])] }]
    else
      [mapping[key], value]
    end
  end
end
