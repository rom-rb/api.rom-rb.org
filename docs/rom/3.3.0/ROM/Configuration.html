def use(plugin, options = {})
  if plugin.is_a?(Array)
    plugin.each { |p| use(p) }
  elsif plugin.is_a?(Hash)
    plugin.to_a.each { |p| use(*p) }
  else
    ROM.plugin_registry.configuration.fetch(plugin).apply_to(self, options)
  end

  self
end
