def command(options)
  case options
  when Symbol
    name = options
    if mappers.key?(name)
      commands[name].with(mappers: mappers[name])
    else
      commands[name]
    end
  when Array
    graph = Commands::Graph.build(commands, options)
    name = graph.name

    if mappers.key?(name)
      graph.with(mappers: mappers[graph.name])
    else
      graph
    end
  else
    raise ArgumentError, "#{self.class}#command accepts a symbol or an array"
  end
end

def relation(name, &block)
  relation =
    if block
      yield(relations[name])
    else
      relations[name]
    end

  if mappers.key?(name)
    relation.to_lazy(mappers: mappers[name])
  else
    relation.to_lazy
  end
end
