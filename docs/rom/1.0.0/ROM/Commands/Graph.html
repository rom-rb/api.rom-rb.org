def call(*args)
  left = root.call(*args)

  right = nodes.map { |node|
    response =
      if node.lazy?
        node.call(args.first, left)
      else
        node.call(left)
      end

    if node.one? && !node.graph?
      [response]
    else
      response
    end
  }

  if one?
    [[left], right]
  else
    [left, right]
  end
end
