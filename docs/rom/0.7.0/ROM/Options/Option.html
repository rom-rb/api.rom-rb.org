def initialize(name, options = {})
  @name = name
  @type = options.fetch(:type) { Object }
  @reader = options.fetch(:reader) { false }
  @allow = options.fetch(:allow) { [] }
  @default = options.fetch(:default) { Undefined }
  @ivar = :"@#{name}" if @reader
end
