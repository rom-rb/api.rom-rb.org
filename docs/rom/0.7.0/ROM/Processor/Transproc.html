def to_transproc
  compose(EMPTY_FN) do |ops|
    combined = header.combined
    ops << t(:combine, combined.map(&method(:combined_args))) if combined.any?
    ops << header.groups.map { |attr| visit_group(attr, true) }
    ops << t(:map_array, row_proc) if row_proc
  end
end
