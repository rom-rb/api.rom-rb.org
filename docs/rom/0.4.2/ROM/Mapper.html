def self.build(header, model)
  klass =
    if header.any? { |attribute| attribute.embedded? }
      Recursive
    elsif header.any? { |attribute| attribute.aliased? }
      Basic
    else
      self
    end

  loader = Proc.new { |tuple, m| m ? m.new(tuple) : tuple }

  klass.new(header, model, loader)
end
