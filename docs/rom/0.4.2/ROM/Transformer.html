def self.build(header)
  operations = header.map do |attribute|
    type = attribute.type

    if type == Hash
      Wrap.new(attribute)
    elsif type == Array
      Group.new(attribute)
    end
  end.compact

  sorted_ops =
    operations.select { |op| Group === op } +
    operations.select { |op| Wrap === op }

  new(sorted_ops.flatten)
end
