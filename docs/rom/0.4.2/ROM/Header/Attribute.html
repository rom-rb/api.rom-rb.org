def self.coerce(input)
  if input.kind_of?(self)
    input
  else
    name = input[0]
    meta = (input[1] || {}).dup

    meta[:type] ||= Object

    if meta.key?(:header)
      meta[:header] = Header.coerce(meta[:header])
    end

    self[meta[:type]].new(name, meta)
  end
end
