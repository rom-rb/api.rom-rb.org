def command(name)
  if mappers.key?(name)
    commands[name].with(mappers: mappers[name])
  else
    commands[name]
  end
end

def relation(name, &block)
  relation =
    if block
      yield(relations[name])
    else
      relations[name]
    end

  if mappers.key?(name)
    relation.to_lazy(mappers: mappers[name])
  else
    relation.to_lazy
  end
end
