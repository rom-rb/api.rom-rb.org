def Coercible.JSONHash(symbol_keys: false, type: Types::Hash)
  Types.Constructor(type) do |value|
    begin
      ::JSON.parse(value.to_s, symbolize_names: symbol_keys)
    rescue ::JSON::ParserError
      value
    end
  end
end
