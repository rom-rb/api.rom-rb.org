def self.build(uri)
  require_adapter(uri)

  klass = get(uri)

  if klass.name == 'Axiom::Adapter::Memory'
    klass.new
  else
    klass.new(uri)
  end
end

def self.get(uri)
  uri_scheme = uri.scheme

  REGISTRY.fetch(uri_scheme) {
    raise(
      UnknownAdapterError,
      "#{uri_scheme.inspect} is no registered uri scheme"
    )
  }
end
