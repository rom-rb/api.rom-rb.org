def identity(object)
  keys = mapper.identity(object)

  if keys.empty?
    object.__id__
  else
    keys
  end
end

def load(tuple)
  identity = mapper.identity_from_tuple(tuple)
  identity_map.fetch_object(identity) { load_and_track(identity, tuple) }
end
