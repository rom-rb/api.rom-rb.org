def command(name, options = {}, &block)
  relation = @restriction.relation
  key = options[:from] || relation
  proc = @restriction.proc

  if @parent_node
    @parent_node.command(name, relation, key, proc, &block)
  else
    super(name, relation, key, proc, &block)
  end
end
