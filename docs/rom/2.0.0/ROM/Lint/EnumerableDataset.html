def lint_each
  result = []
  dataset.each do |tuple|
    result << tuple
  end
  return if result == data

  complain "#{dataset.class}#each must yield tuples"
end

def lint_to_a

  complain "#{dataset.class}#to_a must cast dataset to an array"
