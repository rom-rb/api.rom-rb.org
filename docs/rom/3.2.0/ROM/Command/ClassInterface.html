def create_class(name, type, &block)
  klass = Dry::Core::ClassBuilder
    .new(name: "#{Dry::Core::Inflector.classify(type)}[:#{name}]", parent: type)
    .call

  if block
    yield(klass)
  else
    klass
  end
end
