def [](*args)
  if args.size.equal?(1)
    command = super
    mapper = options[:mapper]

    if mapper
      command.curry >> mapper
    else
      command
    end
  else
    cache.fetch_or_store(args.hash) { compiler.(*args) }
  end
end
