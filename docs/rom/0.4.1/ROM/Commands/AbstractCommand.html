def initialize(relation, options)
  @relation = relation
  @options = options

  @result = options[:result] || :many

  if !VALID_RESULTS.include?(result)
    raise InvalidOptionError.new(:result, VALID_RESULTS)
  end
end

def call(*args)
  tuples = execute(*args)

  if result == :one
    tuples.first
  else
    tuples
  end
end
