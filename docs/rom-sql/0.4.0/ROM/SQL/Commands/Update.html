def execute(tuple)
  attributes = input[tuple]
  validator.call(attributes)

  changed = diff(attributes.to_h)

  if changed.any?
    update(changed)
  else
    []
  end
end

def update(tuple)
  pks = relation.map { |t| t[primary_key] }
  dataset = relation.dataset
  dataset.update(tuple)
  dataset.unfiltered.where(primary_key => pks).to_a
end
