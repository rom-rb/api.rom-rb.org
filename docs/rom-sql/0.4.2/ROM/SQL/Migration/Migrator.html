def create_file(name, version = generate_version)
  filename = "#{version}_#{name}.rb"
  dirname = Pathname(path)
  fullpath = dirname.join(filename)

  FileUtils.mkdir_p(dirname)
  File.write(fullpath, migration_file_content)

  fullpath
end
