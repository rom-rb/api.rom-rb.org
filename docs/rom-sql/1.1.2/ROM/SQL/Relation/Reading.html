def group(*args, &block)
  if block
    if args.size > 0
      group(*args).group_append(&block)
    else
      new(dataset.__send__(__method__, *schema.group(&block)))
    end
  else
    new(dataset.__send__(__method__, *schema.project(*args).canonical))
  end
end
