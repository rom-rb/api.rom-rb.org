def execute(tuples)
  insert_tuples = with_input_tuples(tuples) do |tuple|
    attributes = input[tuple]
    attributes.to_h
  end

  if insert_tuples.length > 1
    multi_insert(insert_tuples)
  else
    insert(insert_tuples)
  end
end
