def for_combine(spec)
  source_key, target_key, target =
    case spec
    when ROM::SQL::Association
      [*spec.join_keys(__registry__).flatten, spec.call(__registry__)]
    else
      [*spec.flatten, self]
    end

  target.preload(source_key, target_key)
end
