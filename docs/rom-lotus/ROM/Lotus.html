def self.setup(app)
  app.synchronize do
    @config = Config.new(app.configuration.root.join('../..').realpath)

    yield(@config)

    ROM.setup(config.repositories)

    %w(relations mappers commands).each do |component|
      Dir["#{config.root}/lib/*/#{component}/**/*.rb"].each do |file|
        require file
      end
    end

    ROM.finalize
  end
end
