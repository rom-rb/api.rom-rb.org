def dataset(name)
  rql.db(options[:db]).table(name.to_s).run(connection)
  datasets[name] = Dataset.new(rql.table(name.to_s), rql, connection)
end

def dataset?(name)
  rql.db(options[:db]).table(name.to_s).run(connection)
  true
rescue ::RethinkDB::RqlRuntimeError
  false
end
