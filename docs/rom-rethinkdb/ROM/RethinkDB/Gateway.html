def initialize(options)
  @datasets = DatasetCache.new
  @options = options
  @rql = ::RethinkDB::RQL.new
  pool_size = options.fetch(:pool_size, 5).to_i
  options.freeze
  @connections = ConnectionPool.new(size: pool_size) { rql.connect(options).auto_reconnect(true) }
end
